    for(Lead singleLead:Trigger.New){
       singleLead.FirstName= singleLead.FirstName.ToUpperCase();
       singleLead.LastName= singleLead.LastName.ToUpperCase();
       singleLead.Company= singleLead.Company.ToUpperCase();
    }
        
    }
